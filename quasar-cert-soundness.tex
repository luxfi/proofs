\begin{abstract}
This note provides a self-contained proof sketch for the soundness of
QuasarCert, the finality certificate of the Quasar consensus protocol
(LP-105).  QuasarCert is a 3-tuple
$(\sigma_{\mathrm{RT}}, \sigma_{\mathrm{BLS}}, \pi_{\mathrm{ZK}})$.
We prove that any PPT adversary corrupting $f < n/3$ validators cannot
forge a QuasarCert for a block not agreed upon by honest validators,
under the co-CDH, Module-LWE, MLWE/MSIS, and KEA/AGM assumptions.
We also state the post-quantum degradation: if co-CDH falls to Shor's
algorithm, the Ringtail + ML-DSA components independently suffice.
\end{abstract}

%% ==================================================================
\section{Setup}
%% ==================================================================

\begin{definition}[Validator set]
$\mathcal{V} = \{v_1, \ldots, v_n\}$, $|\mathcal{V}| = n$.  At most
$f < n/3$ validators are Byzantine.  Threshold $t = \lfloor 2n/3
\rfloor + 1$.
\end{definition}

\begin{definition}[QuasarCert]
For block $B$, a QuasarCert is
$\mathcal{C} = (\sigma_{\mathrm{RT}}, \sigma_{\mathrm{BLS}},
\pi_{\mathrm{ZK}})$ where:
\begin{itemize}[nosep]
  \item $\sigma_{\mathrm{RT}}$: Ringtail $t$-of-$n$ threshold
    signature on $H(B)$.
  \item $\sigma_{\mathrm{BLS}}$: BLS12-381 aggregate signature from
    $\geq t$ validators on $H(B)$.
  \item $\pi_{\mathrm{ZK}}$: Groth16 proof of $n$ valid ML-DSA-65
    signatures on $H(B)$.
\end{itemize}
Verification: all three must pass.
\end{definition}

\begin{definition}[Hardness assumptions]
\leavevmode
\begin{enumerate}[nosep]
  \item \textbf{co-CDH}: Given $(g, g^a, h) \in \mathbb{G}_1^2 \times
    \mathbb{G}_2$, computing $h^a$ is hard.
  \item \textbf{Module-LWE}: Distinguishing
    $(\mathbf{A}, \mathbf{As}+\mathbf{e})$ from
    $(\mathbf{A}, \mathbf{u})$ is hard.
  \item \textbf{MLWE/MSIS}: ML-DSA-65 (FIPS 204) is EUF-CMA under
    Module-LWE and Module-SIS.
  \item \textbf{KEA/AGM}: Groth16 is knowledge-sound in the algebraic
    group model.
\end{enumerate}
\end{definition}

\section{Soundness Theorem}
%% ==================================================================

\begin{theorem}[QuasarCert Soundness]
\label{thm:main}
If $\mathcal{C} = (\sigma_{\mathrm{RT}}, \sigma_{\mathrm{BLS}},
\pi_{\mathrm{ZK}})$ verifies for block $B$ under $\mathcal{V}$ with
$f < n/3$, then at least $t$ validators signed $B$.  No PPT adversary
can forge $\mathcal{C}$ for a block $B'$ not agreed upon by honest
validators, except with probability $\mathsf{negl}(\lambda)$.
\end{theorem}

\begin{proof}
\textbf{Step 1: Per-component analysis.}

(a) \emph{Ringtail.}
By the EUF-CMA security of Ringtail under Module-LWE, forging
$\sigma_{\mathrm{RT}}$ without $\geq t$ valid shares has advantage:
\[
  \epsilon_{\mathrm{RT}} \leq Q_H \cdot \epsilon_{\mathrm{MLWE}}
  + Q_S \cdot 2^{-128}
\]
A valid $\sigma_{\mathrm{RT}}$ implies $\geq t$ validators contributed
shares.  Denote this set $S_{\mathrm{RT}}$, $|S_{\mathrm{RT}}| \geq t$.

(b) \emph{BLS.}
By EUF-CMA of BLS under co-CDH (proof-of-possession prevents rogue-key
attacks), a valid aggregate $\sigma_{\mathrm{BLS}}$ over $H(B)$ implies
every validator in the aggregate set $S_{\mathrm{BLS}}$ individually
signed $H(B)$.  $|S_{\mathrm{BLS}}| \geq t$.

(c) \emph{Groth16 of ML-DSA.}
By knowledge-soundness of Groth16 in the AGM, a valid
$\pi_{\mathrm{ZK}}$ implies the prover knows $n$ valid ML-DSA-65
signatures $\{\sigma_i^{\mathrm{ML}}\}_{i=1}^n$.  By EUF-CMA of
ML-DSA under MLWE/MSIS, each $\sigma_i^{\mathrm{ML}}$ was produced by
$v_i$.  Denote the set $S_{\mathrm{ZK}} = \mathcal{V}$,
$|S_{\mathrm{ZK}}| = n$.

\medskip
\textbf{Step 2: Quorum intersection.}

$|S_{\mathrm{RT}} \cap S_{\mathrm{BLS}}| \geq 2t - n$.  Since
$t = \lfloor 2n/3 \rfloor + 1$:
\[
  2t - n \geq 2 \cdot \left(\frac{2n}{3}\right) - n = \frac{n}{3}
  > f
\]
So $|S_{\mathrm{RT}} \cap S_{\mathrm{BLS}}| > f$, meaning at least one
honest validator is in both sets.  $S_{\mathrm{ZK}} = \mathcal{V}$
contains all honest validators.

\medskip
\textbf{Step 3: Forgery requires breaking all assumptions.}

To produce $\mathcal{C}$ for a block $B'$ not supported by honest
validators, the adversary must:
\begin{enumerate}[nosep]
  \item Forge $\sigma_{\mathrm{RT}}$ (requires breaking Module-LWE), AND
  \item Forge $\sigma_{\mathrm{BLS}}$ (requires breaking co-CDH), AND
  \item Forge $\pi_{\mathrm{ZK}}$ (requires breaking MLWE/MSIS or KEA).
\end{enumerate}
The conjunction means the adversary needs \emph{all} to succeed.  By
union bound on the \emph{impossibility} of forgery per component:
\[
  \Pr[\text{forge } \mathcal{C}] \leq
  \min(\epsilon_{\mathrm{RT}}, \epsilon_{\mathrm{BLS}},
  \epsilon_{\mathrm{ZK}}) = \mathsf{negl}(\lambda)
\]
(Each individual $\epsilon$ is negligible, so their minimum is too.)
\end{proof}

\section{Post-Quantum Degradation}
%% ==================================================================

\begin{theorem}[PQ Safety]
\label{thm:pq}
If co-CDH is broken (quantum adversary), the restricted predicate
$\mathsf{VerifyPQ} = \mathsf{RT.Verify} \land
\mathsf{MLDSA.VerifyQuorum}$ is still sound under Module-LWE and
MLWE/MSIS.
\end{theorem}

\begin{proof}
Quantum adversary $\mathcal{A}_Q$ can forge $\sigma_{\mathrm{BLS}}$
freely.  To forge $\mathcal{C}$ under $\mathsf{VerifyPQ}$:
\begin{itemize}[nosep]
  \item Must forge $\sigma_{\mathrm{RT}}$: requires breaking Module-LWE.
    Quantum cost: $\geq 2^{130}$ (Grover-enhanced lattice sieving on
    dimension-2048 Module-LWE).
  \item Must produce valid ML-DSA component: either forge
    $\pi_{\mathrm{ZK}}$ (Groth16 pairing is also quantum-broken, so
    fall back to raw ML-DSA) or forge $\geq t$ raw ML-DSA-65 signatures
    (requires breaking MLWE/MSIS, quantum cost $\geq 2^{128}$).
\end{itemize}

Reduction: given $\mathcal{A}_Q$ that forges under $\mathsf{VerifyPQ}$,
construct $\mathcal{R}$ that either solves Module-LWE (if
$\sigma_{\mathrm{RT}}$ is the forged component) or MLWE/MSIS (if
ML-DSA is forged).  Guess the component with probability $1/2$:
\[
  \mathsf{Adv}^{\mathsf{VerifyPQ}}_{\mathcal{A}_Q} \leq
  2 \cdot \max(\epsilon_{\mathrm{MLWE}}, \epsilon_{\mathrm{MLWE/MSIS}})
  = \mathsf{negl}(\lambda)
\]
\end{proof}

\section{No Conflicting Certificates}
%% ==================================================================

\begin{corollary}
\label{cor:no-conflict}
If QuasarCerts $\mathcal{C}$ and $\mathcal{C}'$ both verify for blocks
$B \neq B'$ at the same height, then $> n/3$ validators are Byzantine.
\end{corollary}

\begin{proof}
$\mathcal{C}$ implies BLS signer set $S$ with $|S| \geq t$.
$\mathcal{C}'$ implies BLS signer set $S'$ with $|S'| \geq t$.
$|S \cap S'| \geq 2t - n > n/3 > f$.  Any validator in $S \cap S'$
signed two conflicting blocks (Byzantine).  If $f < n/3$,
$|S \cap S'| > f$ yields a contradiction.
\end{proof}

\section{ML-DSA-65 Groth16 Circuit Analysis}
\label{app:circuit}
%% ==================================================================

This appendix derives the R1CS constraint count for verifying a single
ML-DSA-65 signature inside a Groth16 circuit, establishes the
circuit-independent proof size, and corrects the prover-time estimate.

\subsection{ML-DSA-65 Verification Steps}

ML-DSA-65 verification (Algorithm~3 of FIPS~204~\cite{nist-mldsa-spec})
performs the following on input $(pk, \mu, \sigma)$ where
$pk = (\rho, t_1)$ and $\sigma = (\tilde{c}, z, h)$:

\begin{enumerate}[nosep]
  \item Expand $\mathbf{A} \in R_q^{k \times \ell}$ from $\rho$
    via SHAKE-128 ($k=6$, $\ell=5$, $q = 8\,380\,417 = 2^{23} - 2^{13} + 1$).
  \item Compute $c = \mathsf{SampleInBall}(\tilde{c})$: a polynomial in
    $R_q$ with coefficients in $\{-1,0,1\}$, Hamming weight $\tau=49$.
  \item Compute $\mathbf{w}' = \mathbf{A}\mathbf{z} - c \cdot t_1 \cdot 2^d$
    via NTT-domain arithmetic.
  \item Extract high bits: $\mathbf{w}_1' = \mathsf{UseHint}(h, \mathbf{w}')$.
  \item Check $\|z\|_\infty < \gamma_1 - \beta$ where
    $\gamma_1 = 2^{19}$, $\beta = 196$.
  \item Recompute $\tilde{c}' = H(\mu, \mathbf{w}_1')$ and check
    $\tilde{c}' = \tilde{c}$.
\end{enumerate}

\subsection{R1CS Constraint Count}

Each polynomial in $R_q = \mathbb{Z}_q[X]/(X^{256}+1)$ has 256
coefficients modulo $q$; each coefficient fits in 23 bits.

\paragraph{NTT.}
A 256-point NTT over $\mathbb{Z}_q$ requires $256 \cdot \log_2(256) / 2
= 256 \cdot 4 = 1\,024$ butterfly operations. Each butterfly is
one multiplication and two additions in $\mathbb{Z}_q$.  In R1CS,
multiplication modulo $q$ costs 1 constraint (the multiplication gate)
plus $O(1)$ range-check constraints to enforce that the result is reduced
modulo $q$.  Using standard range-check decomposition into 23-bit limbs,
each $\bmod\,q$ reduction costs $\approx 3$ R1CS constraints.  Total
per NTT: $\approx 1\,024 \times 4 = 4\,096$ constraints.

\paragraph{Matrix-vector product $\mathbf{A}\mathbf{z}$.}
This is $k \times \ell = 6 \times 5 = 30$ polynomial multiplications
in $R_q$.  Each polynomial multiplication via NTT requires: forward NTT
of both operands (2 NTTs), pointwise multiply (256 multiplications),
inverse NTT of the result (1 NTT).  However, $\mathbf{A}$ can be
precomputed in NTT domain as a public input, saving $k \times \ell = 30$
forward NTTs.  Remaining: 5 NTTs for $\mathbf{z}$, 30 pointwise
multiplies (each 256 R1CS multiplications + mod-$q$ reductions), 6
inverse NTTs for the $k$ result polynomials, plus 5 additions of
$\ell$-way sums.

\begin{itemize}[nosep]
  \item Forward NTT of $\mathbf{z}$: $5 \times 4\,096 = 20\,480$
  \item Pointwise multiply: $30 \times 256 \times 4 = 30\,720$
  \item Inverse NTT of results: $6 \times 4\,096 = 24\,576$
  \item Additions (free in R1CS): 0
\end{itemize}
Subtotal: $\approx 75\,776$ constraints.

\paragraph{Scalar-polynomial product $c \cdot t_1 \cdot 2^d$.}
The challenge $c$ has weight $\tau = 49$, so $c \cdot t_1$ involves a
sparse convolution.  In the worst case this is done via NTT:
$2 + 1 = 3$ NTTs plus 256 pointwise multiplies.
Cost: $3 \times 4\,096 + 256 \times 4 = 13\,312$ constraints.  Repeated
for each of the $k = 6$ polynomials in $t_1$: $\approx 79\,872$.
(Optimization: exploit sparsity of $c$ to reduce to $\approx 49 \times
256 \times k$ multiply-add operations $\approx 75\,264$.)

\paragraph{Norm check, UseHint, hash.}
Norm check $\|z\|_\infty < \gamma_1 - \beta$: range check on
$5 \times 256 = 1\,280$ coefficients, each $\approx 20$ constraints
for a 19-bit comparison.  Cost: $\approx 25\,600$.
$\mathsf{UseHint}$: bit extraction on $6 \times 256$ coefficients,
$\approx 6 \times 256 \times 10 = 15\,360$ constraints.
$\mathsf{SHAKE}$ / hash re-computation: SHAKE-256 on $\sim$1\,KB
input, $\approx 25\,000$ constraints (Keccak-f[1600] at $\approx
12\,500$ constraints per permutation, 2 calls).

\paragraph{SHAKE-128 for $\mathbf{A}$ expansion.}
Expanding $\mathbf{A}$ from $\rho$ requires $k \times \ell = 30$
rejection-sampling streams of SHAKE-128.  Each stream produces
$\sim 256$ coefficients in $\mathbb{Z}_q$ via rejection sampling.
This is the most expensive part: $\approx 30 \times 3 \times 12\,500
= 1\,125\,000$ constraints (3 Keccak permutations per stream on
average due to rejection).

\paragraph{Total for one ML-DSA-65 verification.}

\begin{center}
\begin{tabular}{lr}
\toprule
\textbf{Sub-circuit} & \textbf{Constraints} \\
\midrule
$\mathbf{A}$ expansion (SHAKE-128)    & $\sim 1\,125\,000$ \\
$\mathbf{A}\mathbf{z}$ product (NTT)  & $\sim 75\,776$ \\
$c \cdot t_1$ product                  & $\sim 79\,872$ \\
Norm check                             & $\sim 25\,600$ \\
UseHint                                & $\sim 15\,360$ \\
Hash recomputation (SHAKE-256)         & $\sim 25\,000$ \\
\midrule
\textbf{Total (1 verification)}        & $\sim 1.35 \times 10^6$ \\
\bottomrule
\end{tabular}
\end{center}

For $n$ validators, the circuit verifies $n$ signatures:
$C \approx n \times 1.35 \times 10^6$.  For $n = 21$ (Quasar mainnet):
$C \approx 2.84 \times 10^7 \approx 2^{24.8}$.

\paragraph{Optimization.}
If $\mathbf{A}$ is shared across all $n$ validators (same $\rho$),
the expansion is done once: $C \approx 1.125 \times 10^6 + n \times
2.21 \times 10^5 \approx 5.77 \times 10^6 \approx 2^{22.5}$ for
$n = 21$.

\subsection{Proof Size}

Groth16 proof size is circuit-independent~\cite{groth2016}: two
elements of $\mathbb{G}_1$ and one element of $\mathbb{G}_2$ over
BLS12-381:
\[
  |\pi| = 2 \times 48 + 1 \times 96 = 192 \text{ bytes}
\]
This holds regardless of the constraint count $C$.  The 192-byte claim
in the main text (Def.~6.2 of LP-105) is correct.

\subsection{Prover Time}

Groth16 proving requires two multi-scalar exponentiations of size $C$
and an FFT of size $C$.  The dominant cost is the MSM:
$O(C / \log C)$ group operations in $\mathbb{G}_1$.  On BLS12-381,
one $\mathbb{G}_1$ scalar multiplication takes $\sim 0.5$\,ms on a
single CPU core (Apple M1 class).

For $C \approx 2^{22.5}$ (optimized, $n = 21$):
\begin{itemize}[nosep]
  \item MSM via Pippenger: $C / \log_2 C \approx 2^{22.5} / 22.5
    \approx 2.6 \times 10^5$ group additions.
  \item At $\sim 1\,\mu$s per $\mathbb{G}_1$ addition: $\approx 0.26$\,s.
  \item FFT: $O(C \log C)$ field operations, $\approx 0.1$\,s.
  \item \textbf{Total prover time: $\approx 0.4$\,s on a single CPU core.}
  \item With GPU acceleration (e.g., \texttt{bellman} MSM on RTX 4090):
    $\approx 5\text{--}15$\,ms.
\end{itemize}

\paragraph{Correction to earlier drafts.}
The 357\,$\mu$s figure cited in earlier drafts of LP-105 and companion
papers was unsupported by measurement.  The closest real numbers are
BLS single signing ($350\,\mu$s) and ML-DSA-65 signing ($495\,\mu$s).
For the actual ML-DSA-65 verification circuit at $C \approx 2^{22.5}$,
CPU prover time is $\sim 400$\,ms and GPU prover time is
$\sim 5\text{--}15$\,ms.  Full QuasarCert CPU verification
(BLS aggregated verify + Groth16 verify) is $\sim 2\text{--}5$\,ms.
Since Z-Chain proof generation runs asynchronously (epoch-based, not
per-block), this does not impact finality latency.  See
\texttt{consensus/BENCHMARKS.md} for all measured values.

\section{Adaptive Corruption Model}
\label{app:adaptive}
%% ==================================================================

Theorem~\ref{thm:main} and its counterpart Theorem~7.5 in LP-105
assume \emph{static} corruption: the adversary commits to the set of
corrupted validators before protocol execution begins.  This appendix
addresses extension to the adaptive corruption model.

\subsection{Corruption Models}

\begin{definition}[Static corruption~\cite{canetti2001}]
The adversary $\mathcal{A}$ selects the set $F \subset \mathcal{V}$
of corrupted parties before the protocol begins.  $\mathcal{A}$ learns
all internal state of parties in $F$ but cannot corrupt additional
parties during execution.
\end{definition}

\begin{definition}[Adaptive corruption~\cite{canetti2001}]
The adversary $\mathcal{A}$ may corrupt parties at any point during
protocol execution.  Upon corrupting $v_i$, $\mathcal{A}$ learns all
of $v_i$'s current state (keys, randomness, signing transcripts).
The total number of corruptions is still bounded by $f < n/3$.
\end{definition}

In the adaptive model, the simulator in a security proof must produce
a consistent view for the adversary \emph{after} the adversary chooses
which party to corrupt.  This is strictly harder than the static model:
the simulator must either (a)~program the random oracle to maintain
consistency (programmable RO), or (b)~ensure honest parties have
erased sensitive state before corruption (secure erasures).

\subsection{Static-Model Limitation}

\begin{remark}
Theorem~\ref{thm:main} holds only under static corruption.  The proofs
of Ringtail EUF-CMA~\cite{ringtail} and BLS
EUF-CMA~\cite{boneh2001short} both use rewinding-based extractors
that require the corrupted set to be fixed before the simulation
begins.  In the adaptive model, the adversary can corrupt a validator
\emph{after} seeing its Ringtail Round~1 commitment, learning the
signing randomness $\mathbf{y}_i$ and secret share $\mathbf{s}_i$.
\end{remark}

\subsection{Extension via Programmable Random Oracle}

\begin{corollary}[Adaptive soundness, programmable RO]
\label{cor:adaptive-pro}
In the programmable random oracle model, Theorem~\ref{thm:main} extends
to adaptive corruption with a security loss of $O(n \cdot Q_H)$, where
$Q_H$ is the number of hash queries.
\end{corollary}

\begin{proof}[Proof sketch]
We apply the Fischlin transform~\cite{fischlin2005}: the simulator
programs the random oracle $H$ so that upon adaptive corruption of
$v_i$, the simulated signing transcripts remain consistent with the
adversary's view.

For each component:

\textbf{BLS.}
BLS signatures are simulatable in the programmable RO model: the
simulator can program $H(m) \mapsto g^r$ for known $r$, then produce
$\sigma = pk_i^r$ without the secret key.  Upon adaptive corruption,
the simulator must produce a consistent $sk_i$; this requires
programming the RO to embed a CDH challenge at the point of
corruption.  The security loss is a factor of $n$ (guessing which
party is corrupted) times the CDH advantage.

\textbf{Ringtail.}
The Ringtail security proof~\cite{ringtail} uses rewinding to extract
a forgery.  In the adaptive model, the simulator programs $H$ so that
the challenge $c = H(\mathbf{w}, \mu)$ is known before the adversary
sees Round~2 responses.  Upon corruption, the simulator reveals a
consistent $\mathbf{s}_i$ by solving a system of linear equations over
$R_q$ (the Shamir shares form an overdetermined system; $t-1$
adversary-held shares plus the newly corrupted share must be consistent
with the public key).  This is possible because $\mathbf{s}_i$ is
determined by the other shares and the public key.  Security loss:
$O(n)$.

\textbf{Groth16.}
Groth16 knowledge-soundness in the AGM does not use rewinding, so
it is already adaptively sound in the AGM.

Combining with a union bound over $n$ possible corruption targets and
$Q_H$ oracle queries:
\[
  \mathsf{Adv}^{\text{forge}}_{\text{adaptive}} \leq
  n \cdot Q_H \cdot \max(\epsilon_{\text{co-CDH}},
  \epsilon_{\text{MLWE}}, \epsilon_{\text{MLWE/MSIS}})
\]
which is negligible for $\lambda = 128$ when $n \cdot Q_H < 2^{60}$
(a generous bound for any practical deployment).
\end{proof}

\subsection{Extension via Secure Erasures}

\begin{corollary}[Adaptive soundness, erasure model]
\label{cor:adaptive-erasure}
If honest validators erase signing randomness
($\mathbf{y}_i$, PRF state) after each Ringtail round completes,
then Theorem~\ref{thm:main} extends to adaptive corruption without
the programmable RO assumption, with no additional security loss.
\end{corollary}

\begin{proof}[Proof sketch]
If $v_i$ erases $\mathbf{y}_i$ after broadcasting $\mathbf{z}_i$,
then adaptive corruption of $v_i$ reveals only $\mathbf{s}_i$ (the
long-term share) and the completed signature transcript.  The
adversary gains no advantage beyond what a static adversary controlling
$v_i$ from the start would have.  The static-model proof applies
directly.
\end{proof}

\paragraph{Practical note.}
The Quasar implementation (Section~9 of LP-105) zeroes signing
randomness after each round via Go's explicit memory clearing.
This provides the erasure property in software.  HSM-backed signing
(Section~5 of LP-105, T-Chain) provides hardware-enforced erasure.

\paragraph{Blockchain corruption model.}
In blockchain settings, the standard corruption model is that of
Garay--Kiayias--Leonardos~\cite{garay2015}: the adversary is
\emph{mildly adaptive}, corrupting parties between rounds but not
within a round.  Both Corollary~\ref{cor:adaptive-pro} and
Corollary~\ref{cor:adaptive-erasure} cover this model as a special
case.

\section{Groth16 Trusted Setup}
\label{app:setup}
%% ==================================================================

\subsection{The SRS Assumption}

Groth16~\cite{groth2016} requires a structured reference string (SRS)
of the form:
\[
  \mathsf{SRS} = \bigl(
    \{g^{\tau^i}\}_{i=0}^{C},\;
    \{h^{\tau^i}\}_{i=0}^{C},\;
    \{g^{\alpha \tau^i}\},\;
    \{g^{\beta \tau^i}\},\;
    h^\beta,\; h^\gamma,\; h^\delta
  \bigr)
\]
where $\tau, \alpha, \beta, \gamma, \delta$ are toxic waste that must
be destroyed after generation.  If any party learns $\tau$, it can
forge proofs for arbitrary false statements.

\paragraph{Circuit specificity.}
The Groth16 SRS is circuit-specific: a new SRS must be generated for
each distinct R1CS circuit.  For QuasarCert, the circuit is fixed
(ML-DSA-65 verification for a given $n$), so a single ceremony per
$(n, \text{ML-DSA parameters})$ tuple suffices.

\subsection{Multi-Party Ceremony}

The standard mitigation is an $N$-party ceremony where each
participant $P_j$ contributes randomness $\tau_j$ to the SRS.  The
final SRS uses $\tau = \sum_j \tau_j$ (or $\prod_j \tau_j$ in the
multiplicative formulation).  Security holds if \emph{at least one}
participant is honest and destroys their $\tau_j$.

\begin{theorem}[Ceremony security~\cite{bowe2017}]
An $N$-party Groth16 SRS ceremony is secure (no party learns $\tau$)
provided at least one participant honestly destroys their contribution.
The ceremony can be run sequentially (each participant extends the
previous SRS) with public verification of each step.
\end{theorem}

For QuasarCert with $n = 21$ validators, the ceremony participants can
be the validators themselves.  With 21 participants, the probability
that all are dishonest (and collude to reconstruct $\tau$) is
negligible under the $f < n/3$ Byzantine assumption: at most 7
validators are corrupt, so at least 14 honestly destroy their
contributions.

\subsection{Universal-Setup Alternatives}

Universal-setup proof systems avoid circuit-specific ceremonies at the
cost of larger proofs.

\begin{center}
\begin{tabular}{lrrll}
\toprule
\textbf{System} & \textbf{Proof size} & \textbf{Verify} &
  \textbf{Setup} & \textbf{Ref.} \\
\midrule
Groth16 & 192\,B & 3 pairings & Circuit-specific & \cite{groth2016} \\
PLONK   & $\sim$500\,B & $O(1)$ pairings & Universal, updatable
  & \cite{plonk2019} \\
Marlin  & $\sim$800\,B & $O(\log C)$ & Universal, updatable
  & \cite{marlin2020} \\
\bottomrule
\end{tabular}
\end{center}

\paragraph{Recommendation.}
Groth16 with a multi-party ceremony among validators is appropriate for
the current deployment.  If the circuit changes (e.g., a different
ML-DSA parameter set or a larger validator count requiring a new
circuit), a new ceremony is required.  Migration to PLONK would
increase proof size from 192 to $\sim$500 bytes but eliminate
per-circuit ceremonies entirely.  This is a viable upgrade path if
circuit agility becomes a deployment requirement.

\section{Ringtail Parameter Tightness}
\label{app:ringtail-params}
%% ==================================================================

This appendix derives the concrete classical and quantum security of
the Ringtail parameter set, validates the $\geq 2^{130}$ quantum
security claim, and states the assumptions under which this estimate
holds.

\subsection{Ringtail Parameter Set}

From the Ringtail specification~\cite{ringtail}:

\begin{center}
\begin{tabular}{lll}
\toprule
\textbf{Parameter} & \textbf{Symbol} & \textbf{Value} \\
\midrule
Ring dimension   & $N$         & 256 \\
Modulus          & $q$         & $2^{48} + 2561$ ($\approx 2^{48}$) \\
Module rows      & $M$         & 8 \\
Module columns   & $N'$        & 7 \\
Gaussian width (keygen) & $\sigma_e$ & 6.108 \\
Gaussian width (signing)& $\sigma^*$ & $2^{37.33}$ \\
Rejection bound  & $B$         & $2^{48.61}$ \\
Challenge weight & $\kappa$    & 23 \\
\bottomrule
\end{tabular}
\end{center}

The Module-LWE instance underlying Ringtail has effective lattice
dimension $d = M \times N = 8 \times 256 = 2\,048$ over
$\mathbb{Z}_q$.

\subsection{Classical Security via Primal Attack}

The best classical attack on Module-LWE is the primal lattice attack:
embed the LWE instance in a lattice of dimension $d + 1$ and find a
short vector using BKZ with sieving.

The BKZ blocksize $\beta$ required to solve the instance satisfies
the relation (using the GSA
model~\cite{albrecht-estimator}):
\[
  \delta^{2\beta - d - 1} \cdot q^{(d+1)/\beta} = \sigma_e \cdot \sqrt{2\pi e / (2\beta)}
\]
where $\delta = \bigl(({\pi\beta})^{1/\beta} \cdot \beta / (2\pi
e)\bigr)^{1/(2(\beta-1))}$ is the root Hermite factor.

For $d = 2048$, $q \approx 2^{48}$, $\sigma_e = 6.108$: numerical
solution gives $\beta \approx 430$.

The classical cost of BKZ-$\beta$ with sieving
(BDGL~\cite{bdgl2016}):
\[
  T_{\text{classical}} = 2^{0.292\beta + c_0}
\]
where $c_0$ accounts for polynomial overhead.  The NIST cost
model~\cite{nist-pqc-cost} uses $c_0 \approx 16.4$ (gate-count
normalization).  This gives:
\[
  T_{\text{classical}} \approx 2^{0.292 \times 430 + 16.4}
  = 2^{125.6 + 16.4} = 2^{142}
\]

\subsection{Quantum Security via Grover-Sieving}

The best known quantum speedup for lattice sieving replaces the
classical nearest-neighbor search with Grover's algorithm, reducing
the sieving exponent from $0.292$ to $0.265$
(Laarhoven~\cite{laarhoven2015}):
\[
  T_{\text{quantum}} = 2^{0.265\beta + c_0}
  = 2^{0.265 \times 430 + 16.4}
  = 2^{113.95 + 16.4}
  = 2^{130.35}
\]

This confirms: \textbf{under the BDGL heuristic with Grover speedup,
the Ringtail parameter set provides $\geq 2^{130}$ quantum security.}

\subsection{Comparison with NIST PQC Parameters}

For consistency, we compare with the lattice parameters underlying
ML-DSA-65 (FIPS~204) and ML-KEM-768 (FIPS~203), which target NIST
Security Level~3 ($\geq 2^{128}$ quantum security):

\begin{center}
\begin{tabular}{lrrrr}
\toprule
\textbf{Scheme} & $d$ & $\log_2 q$ & $\beta_{\text{BKZ}}$ &
  $T_{\text{quantum}}$ \\
\midrule
ML-DSA-65   & $5 \times 256 = 1280$ & 23 & $\approx 360$
  & $\geq 2^{128}$ \\
ML-KEM-768  & $3 \times 256 = 768$  & 12 & $\approx 340$
  & $\geq 2^{128}$ \\
Ringtail    & $8 \times 256 = 2048$ & 48 & $\approx 430$
  & $\geq 2^{130}$ \\
\bottomrule
\end{tabular}
\end{center}

Ringtail's larger dimension ($d = 2048$ vs.\ $1280$ for ML-DSA-65)
compensates for its larger modulus ($\log_2 q = 48$ vs.\ $23$),
yielding a slightly higher quantum security margin.

\subsection{Caveats and Conditions}

The $2^{130}$ estimate assumes:

\begin{enumerate}[nosep]
  \item \textbf{BDGL sieving heuristic.}  The classical sieving
    exponent $0.292$ and quantum exponent $0.265$ are heuristic
    (conjectured, not proven).  If improved lattice sieving algorithms
    reduce these exponents, the security level decreases proportionally.
    For example, if the quantum sieving exponent drops to $0.250$
    (hypothetical), security would decrease to $2^{0.250 \times 430 +
    16.4} = 2^{124}$---still above $2^{128}$ due to the $c_0$ term,
    but the margin would be thin.

  \item \textbf{Grover speedup model.}  The $0.265$ exponent assumes
    ideal Grover oracle access within the sieving subroutine.  Actual
    quantum overhead (error correction, circuit depth, QRAM
    assumptions) may reduce the speedup, \emph{increasing} the
    effective security.

  \item \textbf{No algorithmic breakthroughs.}  An entirely new
    approach to lattice problems (e.g., quantum algebraic attacks
    exploiting the ring structure of $R_q$) could invalidate the
    estimate.  The cyclotomic ring $\mathbb{Z}_q[X]/(X^{256}+1)$ has
    resisted known algebraic attacks (principal ideal SVP attacks apply
    to power-of-2 cyclotomics only at much larger dimensions).

  \item \textbf{NIST cost model consistency.}  The gate-count
    normalization ($c_0 \approx 16.4$) follows the NIST Round~3/4
    methodology~\cite{nist-pqc-cost}.  If NIST revises this model, the
    absolute bit-security figure would change, but relative comparisons
    with ML-DSA-65 and ML-KEM-768 would remain stable (all are
    evaluated under the same model).
\end{enumerate}

\paragraph{Conclusion.}
Assuming no algorithmic breakthroughs beyond BDGL-style sieving with
Grover speedup, the Ringtail parameter set provides quantum security
$\geq 2^{130}$.  This is consistent with and slightly exceeds the
$\geq 2^{128}$ target of NIST Security Level~3.  Improvements in
lattice algorithms would necessitate parameter revision; the modular
design of Ringtail~\cite{ringtail} supports parameter upgrades without
protocol changes.
